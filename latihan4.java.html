public class latihan4 {
    public static void main(String[] args) {
        // Mendeklarasikan array integer dengan ukuran 10
        int[] angka = new int[10];

        // Mengisi elemen-elemen array menggunakan loop for
        for (int i = 0; i < angka.length; i++) {
            angka[i] = i * 30;
        }

        // Mencetak elemen-elemen array menggunakan loop for
        System.out.println("Isi Array:");
        for (int i = 0; i < angka.length; i++) {
            System.out.println("Elemen " + (i + 1) + ": " + angka[i]);
        }

        // Menghitung rata-rata elemen dalam array
        int total = 0;
        for (int i = 0; i < angka.length; i++) {
            total += angka[i];
        }
        double rataRata = (double) total / angka.length;
        System.out.println("Rata-rata: " + rataRata);

        // Mencari elemen tertinggi dalam array
        int maksimum = angka[0];
        for (int i = 1; i < angka.length; i++) {
            if (angka[i] > maksimum) {
                maksimum = angka[i];
            }
        }
        System.out.println("Elemen Tertinggi: " + maksimum);
    }
}
