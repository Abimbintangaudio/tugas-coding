public class soal2 {
    public static void main(String[] args){
        int start = 1;
        int end = 100;
        int countOdd = 0;
        int countEven = 0;

        while (start <= end) {
            if (start % 2 == 0) {
                countEven++;
            } else {
                countOdd++;
            }
            start++;
        }

        System.out.println("jumlah bilangan ganjil :" + countOdd);
        System.out.println("jumlah bilangan genap :" + countEven);
    }
}
